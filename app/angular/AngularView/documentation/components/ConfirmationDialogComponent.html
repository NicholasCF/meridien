import { Component, OnInit, Inject } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';

@Component({
  selector: 'app-confirmation-dialog',
  templateUrl: './confirmation-dialog.component.html',
  styleUrls: ['./confirmation-dialog.component.scss']
})
export class ConfirmationDialogComponent implements OnInit {

  constructor(
    public dialogRef: MatDialogRef<ConfirmationDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public dialogSource: any
  ) { }

  ngOnInit(): void {
  }

  closeDialog(result: string){
    this.dialogRef.close({event: result});
  }
}

<h1 mat-dialog-title>Delete {{dialogSource}}</h1>
<mat-label mat-dialog-content>Are you sure you want to remove {{dialogSource}}? This action is irreversible.</mat-label>

<div mat-dialog-actions align="end">
    <button mat-button (click)="closeDialog('yes')">Yes</button>
    <button mat-button (click)="closeDialog('no')">No</button>
</div>
